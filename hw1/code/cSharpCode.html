Code that generates a point, a line a rectangle and a circle

            
                using System;
                using System.Windows.Forms;
                
                public class MainForm : Form
                {
                    public MainForm()
                    {
                        // Set the size of the form
                        this.Size = new System.Drawing.Size(400, 400);
                
                        // Add labels to represent points, lines, rectangles, and circles
                        AddPoint(50, 50);
                        AddLine(50, 100, 200, 100);
                        AddRectangle(50, 150, 100, 50);
                        AddCircle(200, 200, 50);
                    }
                
                    private void AddPoint(int x, int y)
                    {
                        Label point = new Label();
                        point.Location = new System.Drawing.Point(x, y);
                        point.Text = "*";
                        this.Controls.Add(point);
                    }
                
                    private void AddLine(int x1, int y1, int x2, int y2)
                    {
                        Label line = new Label();
                        line.Location = new System.Drawing.Point(x1, y1);
                        int length = (int)Math.Sqrt(Math.Pow(x2 - x1, 2) + Math.Pow(y2 - y1, 2));
                        line.Size = new System.Drawing.Size(length, 1);
                        line.BackColor = System.Drawing.Color.Red;
                        this.Controls.Add(line);
                    }
                
                    private void AddRectangle(int x, int y, int width, int height)
                    {
                        Label rectangle = new Label();
                        rectangle.Location = new System.Drawing.Point(x, y);
                        rectangle.Size = new System.Drawing.Size(width, height);
                        rectangle.BorderStyle = BorderStyle.FixedSingle;
                        rectangle.BackColor = System.Drawing.Color.Blue;
                        this.Controls.Add(rectangle);
                    }
                
                    private void AddCircle(int x, int y, int radius)
                    {
                        Label circle = new Label();
                        circle.Location = new System.Drawing.Point(x - radius, y - radius);
                        circle.Size = new System.Drawing.Size(2 * radius, 2 * radius);
                        circle.BorderStyle = BorderStyle.FixedSingle;
                        circle.BackColor = System.Drawing.Color.Green;
                        circle.Region = new System.Drawing.Region(new System.Drawing.Drawing2D.GraphicsPath());
                        (circle.Region.GetPath(new System.Drawing.Drawing2D.Matrix())).AddEllipse(0, 0, circle.Width, circle.Height);
                        this.Controls.Add(circle);
                    }
                
                    [STAThread]
                    static void Main()
                    {
                        Application.EnableVisualStyles();
                        Application.Run(new MainForm());
                    }
