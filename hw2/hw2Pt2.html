HASHSET

            

                /*HASHSET*/
                
                class HashSet {
                    constructor() {
                    this.set = Object.create(null);
                    }
                
                    // Add an element to the HashSet
                    add(value) {
                    this.set[value] = true;
                    }
                
                    // Remove an element from the HashSet
                    remove(value) {
                    delete this.set[value];
                    }
                
                    // Check if an element exists in the HashSet
                    has(value) {
                    return this.set[value] === true;
                    }
                
                    // Get the size (number of elements) of the HashSet
                    size() {
                    return Object.keys(this.set).length;
                    }
                
                    // Get all values in the HashSet
                    values() {
                    return Object.keys(this.set);
                    }
                }
            

            
SORTEDSET

            

                /*SORTEDSET*/
                
                class SortedSet {
                    constructor() {
                    this.elements = [];
                    }
                
                    // Add an element to the sorted set
                    add(value) {
                    if (!this.has(value)) {
                        this.elements.push(value);
                        this.elements.sort((a, b) => a - b); // Keep the elements sorted
                    }
                    }
                
                    // Remove an element from the sorted set
                    remove(value) {
                    const index = this.elements.indexOf(value);
                    if (index !== -1) {
                        this.elements.splice(index, 1);
                    }
                    }
                
                    // Check if an element exists in the sorted set
                    has(value) {
                    return this.elements.includes(value);
                    }
                
                    // Get the size (number of elements) of the sorted set
                    size() {
                    return this.elements.length;
                    }
                
                    // Get the sorted values in the set
                    values() {
                    return [...this.elements];
                    }
                
                    // Iterate over the elements using a callback function
                    forEach(callback) {
                    this.elements.forEach(callback);
                    }
                
                    // Clear all elements from the sorted set
                    clear() {
                    this.elements = [];
                    }
                
                    // Get the first element in the sorted set
                    first() {
                    if (this.elements.length > 0) {
                        return this.elements[0];
                    }
                    return undefined;
                    }
                
                    // Get the last element in the sorted set
                    last() {
                    const length = this.elements.length;
                    if (length > 0) {
                        return this.elements[length - 1];
                    }
                    return undefined;
                    }
                
                    // Get the elements in the sorted set that are less than a specified value
                    headSet(value) {
                    return this.elements.filter(element => element < value);
                    }
                
                    // Get the elements in the sorted set that are greater than or equal to a specified value
                    tailSet(value) {
                    return this.elements.filter(element => element >= value);
                    }
                
                    // Get a sub-set of elements within a specified range (inclusive)
                    subSet(start, end) {
                    return this.elements.filter(element => element >= start && element <= end);
                    }
                }

 LINKED LIST
 
            
 
                /*LINKED LIST*/
                
                class Node {
                    constructor(value) {
                    this.value = value;
                    this.next = null;
                    }
                }
                
                class LinkedList {
                    constructor() {
                    this.head = null;
                    }
                
                    // Looping
                    forEach(callback) {
                    let current = this.head;
                    while (current) {
                        callback(current.value);
                        current = current.next;
                    }
                    }
                
                    // Add (Append to the end)
                    add(value) {
                    const newNode = new Node(value);
                    if (!this.head) {
                        this.head = newNode;
                    } else {
                        let current = this.head;
                        while (current.next) {
                        current = current.next;
                        }
                        current.next = newNode;
                    }
                    }
                
                    // Insert at a specific position
                    insert(value, index) {
                    if (index < 0) return;
                
                    const newNode = new Node(value);
                
                    if (index === 0) {
                        newNode.next = this.head;
                        this.head = newNode;
                    } else {
                        let current = this.head;
                        let count = 0;
                
                        while (current) {
                        if (count === index - 1) {
                            newNode.next = current.next;
                            current.next = newNode;
                            break;
                        }
                        current = current.next;
                        count++;
                        }
                    }
                    }
                
                    // Remove
                    remove(value) {
                    if (!this.head) return;
                
                    if (this.head.value === value) {
                        this.head = this.head.next;
                        return;
                    }
                
                    let current = this.head;
                    while (current.next) {
                        if (current.next.value === value) {
                        current.next = current.next.next;
                        return;
                        }
                        current = current.next;
                    }
                    }
                
                    // Get
                    get(index) {
                    if (index < 0) return undefined;
                    let current = this.head;
                    let count = 0;
                    while (current) {
                        if (count === index) return current.value;
                        current = current.next;
                        count++;
                    }
                    return undefined;
                    }
                
                    // Check existence of value
                    contains(value) {
                    let current = this.head;
                    while (current) {
                        if (current.value === value) return true;
                        current = current.next;
                    }
                    return false;
                    }
                
                    // Get the size (number of elements) of the linked list
                    size() {
                    let count = 0;
                    let current = this.head;
                    while (current) {
                        count++;
                        current = current.next;
                    }
                    return count;
                    }
                }
